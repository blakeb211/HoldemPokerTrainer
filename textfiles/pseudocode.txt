void AssignPrimeId(ref Hand h)
{
    Debug.Assert(h.ranks.Length == 5 && h.suits.Length == 5);
    
    h.primeId = 1;
    
    for (int i = 0; i < 5; i++)
    {
        h.primeId *= CardRankToPrimeDict( h.ranks[i] );   
    }
      
    return;
}

void AssignHandRank(ref Hand h)
{
    if ( CheckFlush(h) )
    {
    	h.handRank = flushDict( h.primeId );
    	return;
    }
    
    h.handRank = nonFlushDict( h.primeId );
    return;
}
